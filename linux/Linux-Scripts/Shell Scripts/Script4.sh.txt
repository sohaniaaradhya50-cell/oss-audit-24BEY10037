# Usage: ./log_analyzer.sh /var/log/syslog [cite: 169]

LOGFILE=${1:-"/var/log/syslog"} # Default to syslog if no arg provided
KEYWORD=${2:-"error"}           # Default keyword is 'error' [cite: 171]
COUNT=0

if [ ! -f "$LOGFILE" ]; then
    echo "Error: File $LOGFILE not found[cite: 176]."
    exit 1
fi

# Check if file is empty [cite: 184]
if [ ! -s "$LOGFILE" ]; then
    echo "The log file is empty. No analysis possible."
    exit 1
fi

echo "Analyzing $LOGFILE for '$KEYWORD'..."

# Read file line by line [cite: 177]
while IFS= read -r LINE; do
    if echo "$LINE" | grep -iq "$KEYWORD"; then

    fi
done < "$LOGFILE"
