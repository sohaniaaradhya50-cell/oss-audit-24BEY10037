#!/bin/bash
# Script 3: Disk and Permission Auditor

DIRS=("/etc" "/var/log" "/home" "/usr/bin" "/tmp")
# Adding Python-specific directory to audit [cite: 162]
PYTHON_DIR="/usr/lib/python3"

echo "Directory Audit Report"
echo "------------------------------------------"

for DIR in "${DIRS[@]}"; do
    if [ -d "$DIR" ]; then
        # Extract permissions, owner, and group [cite: 157]
        PERMS=$(ls -ld "$DIR" | awk '{print $1, $3, $4}')
        # Get directory size [cite: 157]
        SIZE=$(du -sh "$DIR" 2>/dev/null | cut -f1)
        echo "DIR: $DIR | Permissions: $PERMS | Size: $SIZE"
    else
        echo "ALERT: $DIR does not exist on this system[cite: 161]."
    fi
done

echo "------------------------------------------"
echo "Checking Python Library Directory..."
if [ -d "$PYTHON_DIR" ]; then
    ls -ld "$PYTHON_DIR" | awk '{print "Python Config/Lib Path: "$9 " | Permissions: "$1}'
else
    echo "Python library directory not found in standard location."
fi
