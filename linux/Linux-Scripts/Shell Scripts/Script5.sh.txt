#!/bin/bash
# Script 5: Open Source Manifesto Generator

echo "Answer three questions to generate your manifesto[cite: 192]."
echo "------------------------------------------"

# Interactive questions [cite: 193, 194, 195]
read -p "1. Name one open-source tool you use every day: " TOOL
read -p "2. In one word, what does 'freedom' mean to you? " FREEDOM
read -p "3. Name one thing you would build and share freely: " BUILD

DATE=$(date +'%d %B %Y')
OUTPUT="manifesto_$(whoami).txt"

# Compose the paragraph [cite: 199]
{
    echo "--- OPEN SOURCE MANIFESTO ---"
    echo "Generated on: $DATE"
    echo "By: $(whoami)"
    echo ""
    echo "I believe that software should be a shared resource for all. 
Just as I use $TOOL every day to solve problems, I believe 
the core of technology should be $FREEDOM. In the spirit 
of the Python community, I commit to building $BUILD 
and sharing it freely with the world to stand on the 
shoulders of giants[cite: 58]."
} > "$OUTPUT"

echo "------------------------------------------"
echo "Manifesto saved to $OUTPUT[cite: 201]."
cat "$OUTPUT" # Display the result [cite: 202]
